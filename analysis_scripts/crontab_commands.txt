# Syntax of Crontab
    MIN HOUR DOM MON DOW CMD
    Desc. (in order) = Minute - Hour - Day of the Month - Day of the Week - Command
    * * * * * command 

# Check the Crontab's availability
    systemctl status cron
    # should say 'Active: active (running)
    $ crontab -l 
    $ crontab -e -> edit mode 
    $ echo $EDITOR (should have vim) | to change --> # export $EDITOR=nano 
    # When selecting an editor, chose [1] /bin/nano --> the easiest
    OR $ sudo crontab -u root -e (might need permisssion, cannot) -- need to log in as the root to directly run the automated command

    $ crontab -u [username] -l 

    # Check the Logs:
        $ sudo cat /var/log/syslog | grep cron
        $ sudo cat /var/log/syslog | grep CRON

    $ @monthly [Python's script path] --> scheduled for the beginning of every month
    $ @daily [Python's script path] --> scheduled for a background job every day (daily log file cleanup)
